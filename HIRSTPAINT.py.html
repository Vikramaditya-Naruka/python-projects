import turtle as turtle_module
import random

turtle_module.colormode(255)
tim = turtle_module.Turtle()
tim.speed("fastest")
tim.penup()
tim.hideturtle()

color_list = [(232, 230, 226), (244, 226, 237), (231, 235, 243), (228, 240, 234), (232, 159, 66), (48, 111, 154), (116, 155, 203), (226, 200, 104), (208, 125, 165), (21, 131, 96), (177, 164, 32), (220, 75, 110), (12, 177, 145), (166, 48, 94), (150, 22, 60), (226, 79, 44), (120, 175, 129), (25, 33, 80), (40, 166, 203), (118, 109, 159), (194, 69, 39), (164, 8, 4), (233, 168, 193), (62, 23, 60), (30, 53, 112), (162, 208, 191), (226, 176, 167), (182, 186, 215), (21, 100, 80), (237, 200, 10)]
tim.setheading(255)
tim.forward(300)
tim.setheading(0)
number_of_dots = 100

for dot_count in range(1,number_of_dots+1):
    tim.dot(20,random.choice(color_list))
    tim.forward(50)

    if dot_count % 10 == 0:
        tim.setheading(90)
        tim.forward(50)
        tim.setheading(180)
        tim.forward(500)
        tim.setheading(0)

screen = turtle_module.Screen()
screen.exitonclick()
