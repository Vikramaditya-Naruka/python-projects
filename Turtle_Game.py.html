# Draw a triangle , square ,pentagon ,
#  hexagon , heptagon , octagon , nonagon and decagon.

from turtle import Turtle , Screen
import random
colors =["gray","dark green","orange red","peru","dark magenta","dark violet","gold"]
tim = Turtle()
tim.shape("turtle")
def draw_shape(number_sides):


    angle = 360/number_sides

    tim.color(random.choice(colors))
    for i in range(number_sides):
        tim.forward(100)
        tim.right(angle)


for shape_side_n in range(3,11):
    draw_shape(shape_side_n)











screen = Screen()
screen.exitonclick()
